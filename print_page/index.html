  }

    this.usuarios.push(novo);

  async atualizar(id, dados) {

    this.usuarios[i] = { ...this.usuarios[i], ...dados, id };

    if (i === -1) return false;
    this.usuarios.splice(i, 1);
    return true;
  }

    рџ’Ў Por que o .bind(this)? Quando o Express invoca um mГ©todo de classe como handler de rota (router.get('/', controller.listarTodos)), ele o chama sem o contexto do objeto original вЂ” o this dentro do mГ©todo seria undefined.
